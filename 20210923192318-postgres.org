* Create user and database

#+begin_src sql
  -- as super user
  psql -U postgres
  CREATE USER jesse WITH PASSWORD 'password';
  CREATE DATABASE people WITH OWNER 'jesse';
  -- quit
  \q

  -- connect to people as jesse
   psql people jesse
#+end_src
