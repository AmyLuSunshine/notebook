* Create a branch
#+begin_src 
git branch branch_name
#+end_src

* Rename a branch
#+begin_src 
git branch old_branch_name -m new_branch_name
#+end_src

* Delete a branch
#+begin_src 
git branch -d branch_to_be_deleted
#+end_src
