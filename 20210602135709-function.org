* Arrow functions

** use return with curly braces
#+begin_src js
const arr = [1, 2, 3, 4];
const arr1 = arr.reduce((acc, value) => {return acc + value});
console.log(arr1);
#+end_src

** no need return without curly braces
#+begin_src js
const arr = [1, 2, 3, 4];
const arr1 = arr.reduce((acc, value) => acc + value);
console.log(arr1);
#+end_src
