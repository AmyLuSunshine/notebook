#+begin_src go
  import (
	"time"
	"math/rand"
  )
  // range : [min, max] -> [5, 10]
  min := 5
  max := 10
  rand.Seed(time.Now().UnixNano())
  randomIndex := rand.Intn(max-min+1) + min
#+end_src
