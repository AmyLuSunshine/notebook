  #+begin_src makefile
    CC = gcc
    all: target1 target2

    target1: target1.c lib1.o lib2.o
        $(CC) target1.c lib1.o lib2.o -o target1

    target2: target1.c lib1.o lib2.o
        $(CC) target1.c lib1.o lib2.o -o target2

    lib1.o:
        $(CC) -c lib1.c

    lib2.o:
        $(CC) -c lib2.c

    clean:
        rm *.o 
  #+end_src
