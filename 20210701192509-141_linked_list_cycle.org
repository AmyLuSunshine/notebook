var hasCycle = function(head) {
    if (head === null) return false;

    let slow = head;
    let fast = head.next;

    while (fast !== null) {
	if (slow === fast) return true;
	if (fast.next === null) return false;

	slow = slow.next;
	fast = fast.next.next;
    }
    return false;
};
